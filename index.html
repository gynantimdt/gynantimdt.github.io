gulp.task("copy-vendor-files", function() {
    var vendorStyles = gulp.src("src/styles/vendor/*.css")
      .pipe(gulp.dest("dist/styles/vendor"));

    var vendorScripts = gulp.src("src/scripts/vendor/*.js")
      .pipe(gulp.dest("dist/scripts/vendor"));

    return merge(vendorStyles, vendorScripts);

gulp angular 自动化构建脚本

发表于 2015-08-29   |   分类于 angular   |  

先贴一段代码，有空再来解释。

var args = require('yargs').argv
var gulp = require('gulp')
var pkg = require('./package.json')
var paths = pkg.paths
var _ = require('lodash')
var wiredep = require('wiredep').stream
var del = require('del')

var $ = require('gulp-load-plugins')()

//production
var isProduction = args.p
var imagemin = args.min
var useSourceMaps = false
var lint = true
// Bower wiredep hack
var wiredepOptions = {
  "overrides": {
    "bootstrap": {
      "main": [
        "dist/js/bootstrap.js",
        "dist/css/bootstrap.css",
        "less/bootstrap.less"
      ]
    },
    "font-awesome": {
      "main": [
        "css/font-awesome.css",
        "less/font-awesome.less",
        "scss/font-awesome.scss"
      ]
    }
  }
}

// Es6 files base paths.src.app
var es6FilterFiles = ['es6.js']

// Vendor
var vendorFiles = [paths.src.vendor + 'echarts/build/source/echarts-all.js',
  paths.src.vendor + 'angular-echarts/dist/angular-echarts.js']

var sass = function () {
  return gulp.src(paths.src.sass)
    .pipe($.sass())
    .pipe(gulp.dest('./dist/style'))
}

var vendorPath = paths.src.vendor
var appPath = paths.src.app

gulp.task('watch', function () {
  gulp.watch('./less/**/*.less', function () {
    _.debounce(sass, 1000)
  })
})


// Building scripts
gulp.task('scripts', function () {
  log('Building scripts...')
  var es6Filter = $.filter(es6FilterFiles, { restore: true })

  return gulp.src(paths.src.app + '**/*.js')
    .pipe(es6Filter)
    .pipe($.if(lint, $.eslint()))
    .pipe($.if(lint, $.eslint.format()))
    .pipe($.if(useSourceMaps, $.sourcemaps.init()))
    .pipe($.babel())
    .pipe($.if(useSourceMaps, $.sourcemaps.write()))
    .pipe(es6Filter.restore)
    .pipe(gulp.dest(paths.tmp.app))
})

// Building styles
gulp.task('styles', function () {
  log('Building styles...')
  return gulp.src(paths.src.styles + '*.scss')
    .pipe($.if(useSourceMaps, $.sourcemaps.init()))
    .pipe($.sass())
    .on('error', $.sass.logError)
    .pipe($.if(useSourceMaps, $.sourcemaps.write()))
    .pipe(gulp.dest(paths.tmp.styles));
})

// Copy vendor
gulp.task('copy:vendor', function () {
  log('Copy vendor')
  if (isProduction) {
    return gulp.src(vendorFiles, { base: paths.src.base })
      .pipe(gulp.dest(paths.dest.base))
  } else {
    return gulp.src(paths.src.vendor + '**/*', { base: paths.src.base })
      .pipe(gulp.dest(paths.dest.base))
  }
})


// Inject vendor and app files
gulp.task('inject', ['styles', 'scripts'], function () {
  var injectStyles = gulp.src(paths.tmp.styles + '**/*.css', { read: false })

  var injectScripts = gulp.src(paths.tmp.app + '**/*.js')
    .pipe($.angularFilesort())
    .on('error', handleError)

  var injectOptions = {
    ignorePath: [paths.src.base, paths.tmp.base],
    addRootSlash: false
  }

  return gulp.src(paths.src.base + 'index.html')
    .pipe($.inject(injectStyles, injectOptions))
    .pipe($.inject(injectScripts, injectOptions))
    .pipe(wiredep(wiredepOptions))
    .pipe($.if(isProduction, $.replace('vendor', '../' + paths.src.vendor)))
    .pipe(gulp.dest(paths.tmp.base))
})

// Concatenates and registers AngularJS templates in the $templateCache
gulp.task('partials', ['copy:html'], function () {
  log('generate templateCache')
  return gulp.src(paths.tmp.app + '**/*.html')
    .pipe($.minifyHtml({
      empty: true,
      spare: true,
      quotes: true
    }))
    .pipe($.angularTemplatecache(pkg.name + 'templateCacheHtml.js', {
      module: 'koala',
      root: 'app'
    }))
    .pipe(gulp.dest(paths.tmp.scripts))
})

// Copy images
gulp.task('images', function () {
  log("Copy images to dest...")
  return gulp.src(paths.src.images + "**/*")
    .pipe($.if(imagemin, $.imagemin({
      optimizationLevel: 3,
      progressive: true,
      interlaced: true
    })))
    .pipe(gulp.dest(paths.dest.images))
})

// Copy html to tmp
gulp.task('copy:html', function () {
  log('Copy html to tmp...')
  return gulp.src(paths.src.app + '**/*.html')
    .pipe(gulp.dest(isProduction ? paths.tmp.app : paths.dest.app))
})

// Other file process
gulp.task('copy:other', function () {
  var fileFilter = $.filter(function (file) {
    return file.stat.isFile();
  })
  log('copy other file...')
  return gulp.src([paths.src.base + '*',
    paths.src.css + '**/*',
    '!' + paths.src.base + 'index.html'], {base: paths.src.base})
    .pipe(fileFilter)
    .pipe(gulp.dest(paths.dest.base))
})

// Delete dist and tmp
gulp.task('clean', function (done) {
  log("Delete dist and tmp")
  del([paths.tmp.base, paths.dest.base], done)
})

// Build html
gulp.task('html', ['inject', 'partials'], function () {
  var partialsInjectFile = gulp.src(paths.tmp.scripts + pkg.name + 'templateCacheHtml.js', { read: false });
  var partialsInjectOptions = {
    starttag: '<!-- inject:partials -->',
    ignorePath: paths.tmp.base,
    addRootSlash: false
  };

  var htmlFilter = $.filter('*.html', { restore: true })
  var jsFilter = $.filter('**/*.js', { restore: true })
  var cssFilter = $.filter('**/*.css', { restore: true })
  var assets;

  return gulp.src(paths.tmp.base + 'index.html')
    .pipe($.inject(partialsInjectFile, partialsInjectOptions))
    .pipe(assets = $.useref.assets())
    .pipe($.rev())
    .pipe(jsFilter)
    .pipe($.uglify({ preserveComments: $.uglifySaveLicense })).on('error', handleError)
    .pipe(jsFilter.restore)
    .pipe(cssFilter)
    .pipe($.csso())
    .pipe(cssFilter.restore)
    .pipe(assets.restore())
    .pipe($.useref())
    .pipe($.revReplace())
    .pipe(htmlFilter)
    .pipe($.minifyHtml({
      empty: true,
      spare: true,
      quotes: true,
      conditionals: true,
      loose: true
    }))
    .pipe(htmlFilter.restore)
    .pipe(gulp.dest(paths.dest.base))
    .pipe($.size({ title: '/', showFiles: true }));

})

// Copy tmp
gulp.task('copy:tmp', ['scripts', 'styles', 'inject'], function () {
  log('Copy tmp...')
  return gulp.src([paths.tmp.base + '**/*.js', paths.tmp.base + '**/*.css', paths.tmp.base + '**/*.html'])
    .pipe(gulp.dest(paths.dest.base))
})

// Live
gulp.task('live', ['inject', 'styles', 'scripts', 'images', 'copy:vendor', 'copy:other', 'copy:tmp'], function () {
  log('liveing...')
})

// Log
function log(msg) {
  $.util.log($.util.colors.blue(msg));
}

// Error handler
function handleError(err) {
  log(err.toString());
  this.emit('end');
}

angular url去除#号

发表于 2015-08-26   |   分类于 angular   |  

使用HTML5 Modes去除#号（官方文档），在config中配置

$locationProvider.html5Mode(false)

在head中增加一个标签

< base href="/">

仅在支持HTML5 history API的浏览器有用，否则，url仍然会有#号。
不过这样做会导致不能直接访问angular路由控制的url。例如直接访问 http://foo.com/bar，会报错。浏览器没有办法确定这个是访问由angular控制的路由。因此，需要配置将这些请求转发到index.html上。angular.js官方网站配置转发使用的Node中间件源码如下：

//rewrite connect middleware
rewrite: function(){
  return function(req, res, next){
    var REWRITE = /\/(guide|api|cookbook|misc|tutorial|error).*$/,
        IGNORED = /(\.(css|js|png|jpg|gif)$|partials\/.*\.html$)/,
        match;

    if (!IGNORED.test(req.url) && (match = req.url.match(REWRITE))) {
      console.log('rewriting', req.url);
      req.url = req.url.replace(match[0], '/index.html');
    }
    next();
  };
}

JavaScript中类型检测

发表于 2015-07-09   |   分类于 JavaScript   |  

javascript各类型之间的关系

  1. 内置（Build-in）对象与原生（Naitve）对象的区别在于：前者总是在引擎初始化阶段就被创建好的对象，是后者的一个子集；而后者包括了一些在运行过程中动态创建的对象。
  2. 引擎扩展对象是一个并不太大的集合，一般来说比较确定，它们也属于引擎的原生对象（但不属于ECMA规范的原生对象）。
  3. 宿主对象不是引擎的原生对象，而是由宿主框架通过某种机制注册到JavaScript引擎中的对象。
  4. 一些宿主会把自己提供的对象／构造器也称为“原生对象”，例如Internet Explorer 7就把它提供的XMLHttpRequest()称为原生的——与此相对的是在它的更早先版本中通过“new ActiveXObject(‘Microsoft.XMLHTTP’)”这样的方法创建的对象。

    instanceof、constructor

    instanceof和constructor的区别在于’inherited’。a
  • constructor是obj的原型链上的一个属性。constructor可以被改变(native constructor除外)。

  • instanceof运算符用来检测constructor.prototype是否存在于参数object的原型链上。（Math, JSON and arguments）没有构造函数）

    function C(){}
function D(){}

var o = new C();

// true, because: Object.getPrototypeOf(o) === C.prototype
o instanceof C;

o.constructor === C;// true

// false, because D.prototype is nowhere in o's prototype chain
o instanceof D;

o instanceof Object; // true, because:
C.prototype instanceof Object // true

C.prototype = {};
var o2 = new C();

o2 instanceof C; // true
o2.constructor === Object //true

// false, because C.prototype is nowhere in
// o's prototype chain anymore
// C.prototype 不在o's的原型链上
o instanceof C;

D.prototype = new C(); // use inheritance
var o3 = new D();
o3 instanceof D; // true
o3 instanceof C; // true
